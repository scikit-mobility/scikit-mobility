188 def to_flowdataframe(self, tessellation, self_loops=True): 

189 """ 

190 Aggregate a TrajDataFrame into a FlowDataFrame. 

191 The points that do not have a corresponding tile in the spatial tessellation are removed. 

192  

193 Parameters 

194 ---------- 

195 tessellation : GeoDataFrame 

196 the spatial tessellation to use to aggregate the points. 

197  

198 self_loops : boolean 

199 if True, it counts movements that start and end in the same tile. The default is `True`. 

200  

201 Returns 

202 ------- 

203 FlowDataFrame 

204 the FlowDataFrame obtained as an aggregation of the TrajDataFrame 

205  

206 Examples 

207 -------- 

208 >>> import skmob 

209 >>> from skmob.tessellation import tilers 

210 >>> import pandas as pd 

211 >>> from skmob.preprocessing import filtering 

212 >>> # read the trajectory data (GeoLife, Beijing, China) 

213 >>> url = skmob.utils.constants.GEOLIFE_SAMPLE 

214 >>> df = pd.read_csv(url, sep=',', compression='gzip') 

215 >>> tdf = skmob.TrajDataFrame(df, latitude='lat', longitude='lon', user_id='user', datetime='datetime') 

216 >>> print(tdf.head()) 

217 lat lng datetime uid 

218 0 39.984094 116.319236 2008-10-23 05:53:05 1 

219 1 39.984198 116.319322 2008-10-23 05:53:06 1 

220 2 39.984224 116.319402 2008-10-23 05:53:11 1 

221 3 39.984211 116.319389 2008-10-23 05:53:16 1 

222 4 39.984217 116.319422 2008-10-23 05:53:21 1 

223 >>> # build a tessellation over the city 

224 >>> tessellation = tilers.tiler.get("squared", base_shape="Beijing, China", meters=15000) 

225 >>> # counting movements that start and end in the same tile 

226 >>> fdf = tdf.to_flowdataframe(tessellation=tessellation, self_loops=True) 

227 >>> print(fdf.head()) 

228 origin destination flow 

229 0 49 49 788 

230 1 49 62 1 

231 2 50 50 4974 

232 3 50 63 1 

233 4 61 61 207 

234  

235 See Also 

236 -------- 

237 FlowDataFrame 

238 """ 

239 

240 # Step 1: order the dataframe by user_id, traj_id, datetime 

241 self.sort_values(by=self.__operate_on(), ascending=True, inplace=True) 

242 

243 # Step 2: map the trajectory onto the tessellation 

244 flow = self.mapping(tessellation, remove_na=False) 

245 

246 # Step 3: groupby tile_id and sum to obtain the flow 

247 flow.loc[:, constants.DESTINATION] = flow[constants.TILE_ID].shift(-1) 

248 # excluding rows with points of different users 

249 flow.loc[:, 'next_uid'] = flow[constants.UID].shift(-1) 

250 flow = flow.loc[flow['uid'] == flow['next_uid']] 

251 

252 flow = flow.groupby([constants.TILE_ID, constants.DESTINATION], dropna=True).size().reset_index(name=constants.FLOW) 

253 flow.rename(columns={constants.TILE_ID: constants.ORIGIN}, inplace=True) 

254 

255 if not self_loops: 

256 flow = flow[flow[constants.ORIGIN] != flow[constants.DESTINATION]] 

257 

258 return FlowDataFrame(flow, tessellation=tessellation)

268 """

315 if all(isinstance(x, Polygon) for x in tessellation.geometry):
