34def read(file): 

35 """ 

36 Read a TrajDataFrame from a json file. 

37 

38 :param file: str 

39 path and name of the `json` file to read. 

40 

41 :return: 

42 object loaded from file. 

43 

44 """ 

45 with open(file, 'r') as f: 

46 df_type, json_df, parameters, datetime_columns = json.load(f) 

47 

48 if 'TrajDataFrame' in df_type: 

49 tdf = TrajDataFrame(pd.read_json(json_df, orient='split', date_unit='s', 

50 convert_dates=datetime_columns), parameters=parameters) 

51 return tdf 

52 elif 'FlowDataFrame' in df_type: 

53 return None 

54 else: 

55 print('DataFrame type not recognised.') 

56 return None

82def load_geolife_trajectories(path_to_geolife_data_dir, user_ids=[], 

83 filter_kwargs={'max_speed_kmh': 400}, 

84 compress_kwargs={'spatial_radius_km': 0.2}): 

85 """ 

86 Load the Geolife trajectories in a TrajDataFrame 

87 

88 :param path_to_geolife_data_dir: str 

89 local path of the directory 'Geolife Trajectories 1.3/' 

90 

91 :param user_ids: list 

92 list of user IDs to load. If empty all users are loaded. 

93 

94 :param filter_kwargs: dict 

95 arguments of `preprocessing.filtering.filter()`. If empty, data is not filtered. 

96 

97 :param compress_kwargs: dict 

98 arguments of `preprocessing.compression.compress()`. If empty, data is not compressed. 

99 

100 :return: TrajDataFrame 

101 a TrajDataFrame containing all trajectories 

102 :rtype: TrajDataFrame 

103 

104 """ 

105 tdf = TrajDataFrame(pd.DataFrame()) 

106 

107 path = path_to_geolife_data_dir + 'Data/' 

108 if len(user_ids) == 0: 

109 user_ids = os.listdir(path) 

110 

111 for uid in tqdm(user_ids): 

112 try: 

113 all_files = fnmatch.filter(os.listdir(path + '%s/Trajectory/' % uid), "*.plt") 

114 

115 except NotADirectoryError: 

116 continue 

117 

118 dfg = (pd.read_csv(path + '%s/Trajectory/' % uid + f, 

119 skiprows=6, header=None, usecols=[0, 1, 5, 6]) for f in all_files) 

120 df = pd.concat(dfg, ignore_index=True) 

121 df['datetime'] = df[5] + ' ' + df[6] 

122 df.drop(columns=[5, 6], inplace=True) 

123 df['uid'] = [str(uid) for _ in range(len(df))] 

124 

125 tdf0 = TrajDataFrame(df, latitude=0, longitude=1) 

126 

127 if len(filter_kwargs) > 0: 

128 tdf0 = filtering.filter(tdf0, **filter_kwargs) 

129 

130 if len(compress_kwargs) > 0: 

131 tdf0 = compression.compress(tdf0, **compress_kwargs) 

132 

133 tdf = tdf.append(tdf0) 

134 tdf.parameters = tdf0.parameters 

135 

136 return tdf

139def load_google_timeline(file, user_id=None, min_accuracy_meters=None): 

140 """ 

141 Load a Google Timeline trajectory in a TrajDataFrame. 

142 

143 :param file: path to the file "Location History.json" inside the Google Takeout archive 

144 :type file: str 

145 

146 :param user_id: ``str`` or ``int`` 

147 name or ID of the user 

148 

149 :param float min_accuracy_meters: 

150 remove points with "accuracy" value higher than ``min_accuracy_meters`` meters 

151 

152 :return: TrajDataFrame 

153 :rtype: TrajDataFrame 

154 

155 """ 

156 # Read file 

157 with open(file, 'r') as f: 

158 goog = json.load(f) 

159 df = pd.DataFrame.from_dict(goog['locations']) 

160 

161 # filter out inaccurate points 

162 if min_accuracy_meters is not None: 

163 df = df[df['accuracy'] < min_accuracy_meters] 

164 

165 if user_id is not None: 

166 df['uid'] = [user_id for _ in range(len(df))] 

167 df['latitudeE7'] = df['latitudeE7'] / 1e7 

168 df['longitudeE7'] = df['longitudeE7'] / 1e7 

169 df['timestampMs'] = df['timestampMs'].astype(float) / 1e3 

170 

171 tdf = TrajDataFrame(df, latitude='latitudeE7', longitude='longitudeE7', 

172 datetime='timestampMs', timestamp=True) 

173 

174 return tdf.sort_by_uid_and_datetime()
