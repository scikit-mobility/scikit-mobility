49 def _all_risks(self, traj, targets=None, force_instances=False, show_progress=False): 

50 # """ 

51 # Computes risk for all the users in the data. It applies the risk function to every individual in the data. 

52 # If it is not required to compute the risk for the entire data, the targets parameter can be used to select 

53 # a portion of users to perform the calculation on. 

54 # 

55 # Parameters 

56 # ---------- 

57 # traj: TrajectoryDataFrame 

58 # the dataframe against which to calculate risk. 

59 # 

60 # targets : TrajectoryDataFrame or list, optional 

61 # the users_id target of the attack. They must be compatible with the trajectory data. Default values is None 

62 # in which case risk is computed on all users in traj. The default is `None`. 

63 # 

64 # force_instances : boolean, optional 

65 # if True, returns all possible instances of background knowledge 

66 # with their respective probability of reidentification. The default is `False`. 

67 # 

68 # show_progress : boolean, optional 

69 # if True, shows the progress of the computation. The default is `False`. 

70 # 

71 # Returns 

72 # ------- 

73 # DataFrame 

74 # a DataFrame with the privacy risk for each user, in the form (user_id, risk) 

75 # """ 

76 if targets is None: 

77 targets = traj 

78 else: 

79 if isinstance(targets, list): 

80 targets = traj[traj[constants.UID].isin(targets)] 

81 if isinstance(targets, TrajDataFrame) or isinstance(targets, pd.DataFrame): 

82 targets = traj[traj[constants.UID].isin(targets[constants.UID])] 

83 if show_progress: 

84 tqdm.pandas(desc="computing risk") 

85 risks = targets.groupby(constants.UID).progress_apply(lambda x: self._risk(x, traj, force_instances)) 

86 else: 

87 risks = targets.groupby(constants.UID).apply(lambda x: self._risk(x, traj, force_instances)) 

88 if force_instances: 

89 risks = risks.droplevel(1) 

90 risks = risks.reset_index(drop=True) 

91 else: 

92 risks = risks.reset_index(name=constants.PRIVACY_RISK) 

93 return risks

116 def _risk(self, single_traj, traj, force_instances=False): 

117 """ 

118 # Computes the risk of reidentification of an individual with respect to the entire population in the data. 

119 # 

120 # Parameters 

121 # ---------- 

122 # single_traj : TrajectoryDataFrame 

123 # the dataframe of the trajectory of a single individual. 

124 # 

125 # traj : TrajectoryDataFrame 

126 # the dataframe with the complete data. 

127 # 

128 # force_instances : boolean, optional 

129 # if True, returns all possible instances of background knowledge 

130 # with their respective probability of reidentification. The default is `False`. 

131 # 

132 # Returns 

133 # ------- 

134 # float 

135 # the risk for the individual, expressed as a float between 0 and 1 

136 # """ 

137 instances = self._generate_instances(single_traj) 

138 risk = 0 

139 if force_instances: 

140 inst_data = {constants.LATITUDE: list(), constants.LONGITUDE: list(), 

141 constants.DATETIME: list(), constants.UID: list(), 

142 constants.INSTANCE: list(), constants.INSTANCE_ELEMENT: list(), 

143 constants.PROBABILITY: list()} 

144 inst_id = 1 

145 for instance in instances: 

146 prob = 1.0 / traj.groupby(constants.UID).apply(lambda x: self._match(x, instance)).sum() 

147 elem_count = 1 

148 for elem in instance: 

149 inst_data[constants.LATITUDE].append(elem[0]) 

150 inst_data[constants.LONGITUDE].append(elem[1]) 

151 inst_data[constants.DATETIME].append(elem[2]) 

152 inst_data[constants.UID].append(elem[3]) 

153 inst_data[constants.INSTANCE].append(inst_id) 

154 inst_data[constants.INSTANCE_ELEMENT].append(elem_count) 

155 inst_data[constants.PROBABILITY].append(prob) 

156 elem_count += 1 

157 inst_id += 1 

158 return pd.DataFrame(inst_data) 

159 else: 

160 for instance in instances: 

161 prob = 1.0 / traj.groupby(constants.UID).apply(lambda x: self._match(x, instance)).sum() 

162 if prob > risk: 

163 risk = prob 

164 if risk == 1.0: 

165 break 

166 return risk

354 """ 

355 Matching function for the attack. 

356 For a location attack, only the coordinates are used in the matching. 

357 If a trajectory presents the same locations as the ones in the instance, a match is found.

359 

360 Parameters 

361 ---------- 

362 single_traj : TrajectoryDataFrame 

363 the dataframe of the trajectory of a single individual.

366 an instance of background knowledge. 

367

369 ------- 

370 int 

371 1 if the instance matches the trajectory, 0 otherwise. 

372 """ 

373 locs = single_traj.groupby([constants.LATITUDE, constants.LONGITUDE]).size().reset_index(name=constants.COUNT) 

374 inst = pd.DataFrame(data=instance, columns=single_traj.columns) 

375 inst = inst.astype(dtype=dict(single_traj.dtypes)) 

376 inst = inst.groupby([constants.LATITUDE, constants.LONGITUDE]).size().reset_index(name=constants.COUNT + "inst")
