46def compute_distance_matrix(spatial_tessellation, origins): 

47 """ 

48 Compute the matrix of distances between origin locations and all other locations. 

49 

50 Parameters 

51 ---------- 

52 spatial_tessellation : GeoDataFrame 

53 the spatial tessellation. 

54  

55 origins : list 

56 indexes of the locations of origin. 

57  

58 Returns 

59 ------- 

60 distance_matrix : numpy array with distances between locations in `origins` 

61 and all locations in `spatial_tessellation` 

62 """ 

63 coordinates = spatial_tessellation.geometry.apply(utils.get_geom_centroid, args=[True]).values 

64 

65 n = len(spatial_tessellation) 

66 distance_matrix = np.zeros((n, n)) 

67 

68 for id_i in tqdm(origins): 

69 lat_i, lng_i = coordinates[id_i] 

70 for id_j in range(id_i + 1, n): 

71 lat_j, lng_j = coordinates[id_j] 

72 distance = distfunc((lat_i, lng_i), (lat_j, lng_j)) 

73 distance_matrix[id_i, id_j] = distance 

74 distance_matrix[id_j, id_i] = distance 

75 return distance_matrix

78class Gravity: 

79 """Gravity model. 

80  

81 The Gravity model of human migration. In its original formulation, the probability :math:`T_{ij}` of moving from a location :math:`i` to location :math:`j` is defined as [Z1946]_: 

82  

83 .. math:: 

84 T_{ij} \propto \\frac{P_i P_j}{r_{ij}}  

85 

86 where :math:`P_i` and :math:`P_j` are the population of location :math:`i` and :math:`j` and :math:`r_{ij}` is the distance between :math:`i` and :math:`j`. The basic assumptions of this model are that the number of trips leaving location :math:`i` is proportional to its population :math:`P_i`, the attractivity of location :math:`j` is also proportional to :math:`P_j`, and finally, that there is a cost effect in terms of distance traveled. These ideas can be generalized assuming a relation of the type [BBGJLLMRST2018]_: 

87  

88 .. math:: 

89 T_{ij} = K m_i m_j f(r_{ij}) 

90  

91 where :math:`K` is a constant, the masses :math:`m_i` and :math:`m_j` relate to the number of trips leaving location :math:`i` or the ones attracted by location :math:`j`, and :math:`f(r_{ij})`, called *deterrence function*, is a decreasing function of distance. The deterrence function :math:`f(r_{ij})` is commonly modeled with a powerlaw or an exponential form. 

92 

93 **Constrained gravity models**. Some of the limitations of the gravity model can be resolved via constrained versions. For example, one may hold the number of people originating from a location :math:`i` to be a known quantity :math:`O_i`, and the gravity model is then used to estimate the destination, constituting a so-called *singly constrained* gravity model of the form: 

94 

95 .. math:: 

96 T_{ij} = K_i O_i m_j f(r_{ij}) = O_i \\frac{m_i f(r_{ij})}{\sum_k m_k f(r_{ik})}. 

97  

98 In this formulation, the proportionality constants :math:`K_i` depend on the location of the origin and its distance to the other places considered. We can fix also the total number of travelers arriving at a destination :math:`j` as :math:`D_j = \sum_i T_{ij}`, leading to a *doubly-constrained* gravity model. For each origin-destination pair, the flow is calculated as: 

99  

100 .. math:: 

101 T_{ij} = K_i O_i L_j D_j f(r_{ij}) 

102  

103 where there are now two flavors of proportionality constants 

104  

105 .. math:: 

106 K_i = \\frac{1}{\sum_j L_j D_j f(r_{ij})}, L_j = \\frac{1}{\sum_i K_i O_i f(r_{ij})}. 

107 

108 Parameters 

109 ---------- 

110 deterrence_func_type : str, optional 

111 the deterrence function to use. The possible deterrence function are "power_law" and "exponential". The default is "power_law". 

112 

113 deterrence_func_args : list, optional 

114 the arguments of the deterrence function. The default is [-2.0]. 

115 

116 origin_exp : float, optional  

117 the exponent of the origin's relevance (only relevant to globally-constrained model). The default is `1.0`. 

118 

119 destination_exp : float, optional  

120 the explonent of the destination's relevance. The default is 1.0. 

121 

122 gravity_type : str, optional 

123 the type of gravity model. Possible values are "singly constrained" and "globally constrained". The default is "singly constrained". 

124  

125 name : str, optional 

126 the name of the instantiation of the Gravity model. The default is "Gravity model". 

127

294 def _compute_gravity_score(self, distance_matrix, relevances_orig, relevances_dest): 

295 trip_probs_matrix = self._deterrence_func(distance_matrix, *self._deterrence_func_args) 

296 # trip_probs_matrix = np.transpose( 

297 # trip_probs_matrix * relevances ** self.destination_exp) * relevances ** self._origin_exp 

298 trip_probs_matrix = trip_probs_matrix * relevances_dest ** self.destination_exp * \ 

299 np.expand_dims(relevances_orig ** self._origin_exp, axis=1) 

300 # put the NaN and Inf to 0.0 

301 np.putmask(trip_probs_matrix, np.isnan(trip_probs_matrix), 0.0) 

302 np.putmask(trip_probs_matrix, np.isinf(trip_probs_matrix), 0.0) 

303 

304 # put diagonal elements to zero: i.e. exclude intra-location trips (self flows) 

305 np.fill_diagonal(trip_probs_matrix, 0.) 

306 

307 return trip_probs_matrix

309 def generate(self, spatial_tessellation, tile_id_column=constants.TILE_ID, 

310 tot_outflows_column=constants.TOT_OUTFLOW, relevance_column=constants.RELEVANCE, out_format='flows'): 

311 """ 

312 Start the simulation of the Gravity model. 

313  

314 Parameters 

315 ---------- 

316 spatial_tessellation : GeoDataFrame 

317 the spatial tessellation on which to run the simulation. 

318  

319 tile_id_column : str, optional 

320 the column in `spatial_tessellation` of the location identifier. The default is `constants.TILE_ID`. 

321  

322 tot_outflows_column : str, optional 

323 the column in `spatial_tessellation` with the outflow of the location. The default is `constants.TOT_OUTFLOW`. 

324  

325 relevance_column : str, optional 

326 the column in `spatial_tessellation` with the relevance of the location. The default is `constants.RELEVANCE`. 

327  

328 out_format : str, optional 

329 the format of the generated flows. Possible values are "flows" (average flow between two locations), "flows_sample" (random sample of flows), and "probabilities" (probability of a unit flow between two locations). The default is "flows". 

330  

331 Returns 

332 ------- 

333 FlowDataFrame 

334 the flows generated by the Gravity model. 

335 """ 

336 n_locs = len(spatial_tessellation) 

337 relevances = spatial_tessellation[relevance_column].fillna(0).values 

338 self._tile_id_column = tile_id_column 

339 # self._spatial_tessellation = spatial_tessellation 

340 

341 if out_format not in ['flows', 'flows_sample', 'probabilities']: 

342 print('Output format \"%s\" not available. Flows will be used.\n' 

343 'Available output formats are [flows, flows_sample, probabilities]' % out_format) 

344 out_format = "flows" 

345 

346 if 'flows' in out_format: 

347 if tot_outflows_column not in spatial_tessellation.columns: 

348 raise KeyError("The column 'tot_outflows' must be present in the tessellation.") 

349 tot_outflows = spatial_tessellation[tot_outflows_column].fillna(0).values 

350 

351 # the origin locations are all locations 

352 origins = np.arange(n_locs) 

353 

354 # compute the distances between all pairs of locations 

355 distance_matrix = compute_distance_matrix(spatial_tessellation, origins) 

356 

357 # compute scores 

358 trip_probs_matrix = self._compute_gravity_score(distance_matrix, relevances, relevances) 

359 

360 if self._gravity_type == 'globally constrained': # globally constrained gravity model 

361 trip_probs_matrix /= np.sum(trip_probs_matrix) 

362 

363 if out_format == 'flows': 

364 # return average flows 

365 od_matrix = trip_probs_matrix * np.sum(tot_outflows) 

366 return self._from_matrix_to_flowdf(od_matrix, origins, spatial_tessellation) 

367 elif out_format == 'flows_sample': 

368 # generate random fluxes according to trip probabilities 

369 od_matrix = np.reshape(np.random.multinomial(np.sum(tot_outflows), trip_probs_matrix.flatten()), 

370 (n_locs, n_locs)) 

371 return self._from_matrix_to_flowdf(od_matrix, origins, spatial_tessellation) 

372 else: 

373 # return trip_probs_matrix 

374 return self._from_matrix_to_flowdf(trip_probs_matrix, origins, spatial_tessellation) 

375 

376 else: # singly constrained gravity model 

377 # trip_probs_matrix = np.transpose(trip_probs_matrix / np.sum(trip_probs_matrix, axis=0)) 

378 trip_probs_matrix = (trip_probs_matrix.T / np.sum(trip_probs_matrix, axis=1)).T 

379 # put the NaN and Inf to 0.0 

380 np.putmask(trip_probs_matrix, np.isnan(trip_probs_matrix), 0.0) 

381 np.putmask(trip_probs_matrix, np.isinf(trip_probs_matrix), 0.0) 

382 

383 if out_format == 'flows': 

384 od_matrix = (trip_probs_matrix.T * tot_outflows).T 

385 return self._from_matrix_to_flowdf(od_matrix, origins, spatial_tessellation) 

386 elif out_format == 'flows_sample': 

387 # generate random fluxes according to trip probabilities 

388 od_matrix = np.array([np.random.multinomial(tot_outflows[i], trip_probs_matrix[i]) for i in origins]) 

389 return self._from_matrix_to_flowdf(od_matrix, origins, spatial_tessellation) 

390 else: 

391 # return trip_probs_matrix 

392 return self._from_matrix_to_flowdf(trip_probs_matrix, origins, spatial_tessellation)

401 def _update_training_set(self, flow_example): 

402 

403 id_origin = flow_example[constants.ORIGIN] 

404 id_destination = flow_example[constants.DESTINATION] 

405 trips = flow_example[constants.FLOW] 

406 

407 if id_origin == id_destination: 

408 return 

409 

410 try: 

411 coords_origin = self.lats_lngs[self.tileid2index[id_origin]] 

412 weight_origin = self.weights[self.tileid2index[id_origin]] 

413 except KeyError: 

414 print('Missing information for location ' + \ 

415 '"%s" in spatial tessellation. Skipping ...' % id_origin) 

416 return 

417 

418 try: 

419 coords_destination = self.lats_lngs[self.tileid2index[id_destination]] 

420 weight_destination = self.weights[self.tileid2index[id_destination]] 

421 except KeyError: 

422 print('Missing information for location ' + \ 

423 '"%s" in spatial tessellation. Skipping ...' % id_destination) 

424 return 

425 

426 if weight_destination <= 0: 

427 return 

428 

429 dist = distfunc(coords_origin, coords_destination) 

430 

431 if self._gravity_type == 'globally constrained': 

432 sc_vars = [np.log(weight_origin)] 

433 else: # if singly constrained 

434 sc_vars = ci(self.tileid2index[id_origin], len(self.tileid2index)) 

435 

436 if self._deterrence_func_type == 'exponential': 

437 # exponential deterrence function 

438 self.X += [[1.] + sc_vars + [np.log(weight_destination), - dist]] 

439 else: 

440 # power law deterrence function 

441 self.X += [[1.] + sc_vars + [np.log(weight_destination), np.log(dist)]] 

442 

443 self.y += [float(trips)]

445 def fit(self, flow_df, relevance_column=constants.RELEVANCE): 

446 """ 

447 Fit the parameters of the Gravity model to the flows provided in input, using a Generalized Linear Model (GLM) with a Poisson regression [FM1982]_. 

448 

449 Parameters 

450 ---------- 

451 flow_df : FlowDataFrame  

452 the real flows on the spatial tessellation.  

453  

454 relevance_column : str, optional 

455 the column in the spatial tessellation with the relevance of the location. The default is constants.RELEVANCE. 

456 

457 References 

458 ---------- 

459 .. [FM1982] Flowerdew, R. & Murray, A. (1982) A method of fitting the gravity model based on the Poisson distribution. Journal of regional science 22(2), 191-202, https://onlinelibrary.wiley.com/doi/abs/10.1111/j.1467-9787.1982.tb00744.x 

460 

461 """ 

462 self.lats_lngs = flow_df.tessellation.geometry.apply(utils.get_geom_centroid, args=[True]).values 

463 self.weights = flow_df.tessellation[relevance_column].fillna(0).values 

464 self.tileid2index = dict( 

465 [(tileid, i) for i, tileid in enumerate(flow_df.tessellation[constants.TILE_ID].values)]) 

466 

467 self.X, self.y = [], [] # independent (X) and dependent (y) variables 

468 

469 # flow_df.progress_apply(lambda flow_example: self._update_training_set(flow_example), 

470 # axis=1) 

471 flow_df.apply(lambda flow_example: self._update_training_set(flow_example), axis=1) 

472 

473 # Perform GLM fit 

474 poisson_model = GLM(self.y, self.X, family=sm.genmod.families.family.Poisson(link=sm.genmod.families.links.log())) 

475 poisson_results = poisson_model.fit() 

476 

477 # Set best fit parameters 

478 if self._gravity_type == 'globally constrained': 

479 self._origin_exp = poisson_results.params[1] 

480 self._destination_exp = poisson_results.params[2] 

481 self._deterrence_func_args = [poisson_results.params[3]] 

482 else: # if singly constrained 

483 self._origin_exp = 1. 

484 self._destination_exp = poisson_results.params[-2] 

485 self._deterrence_func_args = [poisson_results.params[-1]] 

486 

487 # we delete the instance variables we do not need anymore 

488 del self.X 

489 del self.y
