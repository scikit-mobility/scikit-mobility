60def get_color(k=-2, color_dict=COLOR): 

61 """ 

62 Return a color (gray if k == -1, random if k < -1) 

63 """ 

64 if k < -1: 

65 return np.random.choice(list(color_dict.values())) # color_dict[random.randint(0,20)] 

66 elif k == -1: 

67 return '#808080' # Gray 

68 else: 

69 return color_dict[k % len(color_dict)]

392def plot_diary(cstdf, user, start_datetime=None, end_datetime=None, ax=None, legend=False): 

393 """ 

394 Plot a mobility diary of an individual in a TrajDataFrame. It requires a TrajDataFrame with clusters, output of `preprocessing.clustering.cluster`. The column `constants.CLUSTER` must be present. 

395 

396 Parameters 

397 ---------- 

398 user : str or int 

399 user identifier whose diary should be plotted. 

400 

401 start_datetime : datetime.datetime, optional 

402 only stops made after this date will be plotted. If `None` the datetime of the oldest stop will be selected. The default is `None`. 

403 

404 end_datetime : datetime.datetime, optional 

405 only stops made before this date will be plotted. If `None` the datetime of the newest stop will be selected. The default is `None`. 

406 

407 ax : matplotlib.axes, optional 

408 axes where the diary will be plotted. If `None` a new ax is created. The default is `None`. 

409 

410 legend : bool, optional 

411 If `True`, legend with cluster IDs is shown. The default is `False`. 

412 

413 Returns 

414 ------- 

415 matplotlib.axes 

416 the `matplotlib.axes` object of the plotted diary. 

417 

418 """ 

419 if ax is None: 

420 fig, ax = plt.subplots(figsize=(20, 2)) 

421 

422 if user is None: 

423 df = cstdf 

424 else: 

425 df = cstdf[cstdf[constants.UID] == user] 

426 

427 if len(df) == 0: 

428 raise KeyError("""User id is not in the input TrajDataFrame.""") 

429 

430 # TODO: add warning if days between start_datetime and end_datetime do not overlap with cstdf 

431 if start_datetime is None: 

432 start_datetime = df[constants.DATETIME].min() 

433 elif type(start_datetime) is str: 

434 start_datetime = pd.to_datetime(start_datetime) 

435 if end_datetime is None: 

436 end_datetime = df[constants.LEAVING_DATETIME].max() 

437 elif type(end_datetime) is str: 

438 end_datetime = pd.to_datetime(end_datetime) 

439 

440 current_labels = [] 

441 

442 for idx, row in df.iterrows(): 

443 

444 t0 = row[constants.DATETIME] 

445 t1 = row[constants.LEAVING_DATETIME] 

446 cl = row[constants.CLUSTER] 

447 

448 color = get_color(cl) 

449 if start_datetime <= t0 <= end_datetime: 

450 if cl in current_labels: 

451 ax.axvspan(t0.to_pydatetime(), t1.to_pydatetime(), lw=0.0, alpha=0.75, color=color) 

452 else: 

453 current_labels += [cl] 

454 ax.axvspan(t0.to_pydatetime(), t1.to_pydatetime(), lw=0.0, alpha=0.75, color=color, label=cl) 

455 

456 plt.xlim(start_datetime, end_datetime) 

457 

458 if legend: 

459 handles, labels_str = ax.get_legend_handles_labels() 

460 labels = list(map(int, labels_str)) 

461 # sort them by labels 

462 import operator 

463 hl = sorted(zip(handles, labels), key=operator.itemgetter(1)) 

464 handles2, labels2 = zip(*hl) 

465 

466 ax.legend(handles2, labels2, ncol=15, bbox_to_anchor=(1., -0.2), frameon=0) 

467 

468 ax.set_title('user %s' % user) 

469 

470 return ax

478def plot_flows(fdf, map_f=None, min_flow=0, tiles='cartodbpositron', zoom=6, flow_color='red', opacity=0.5,

482 """ 

483 :param fdf: FlowDataFrame 

484 `FlowDataFrame` to visualize. 

485

506 

507 :param flow_exp: float 

508 weight exponent used in the function to compute the thickness of the flow edges. 

509 

510 :param style_function: lambda function 

511 GeoJson style function. 

512 

513 :param flow_popup: bool 

514 if `True`, when clicking on a flow edge a popup window displaying information on the flow will appear. 

515 

516 :param num_od_popup: int 

517 number of origin-destination pairs to show in the popup window of each origin location. 

518 

519 :param tile_popup: bool 

520 if `True`, when clicking on a location marker a popup window displaying information on the flows 

521 departing from that location will appear. 

522 

523 :param radius_origin_point: float 

524 size of the location markers. 

525 

526 :param color_origin_point: str 

527 color of the location markers. 

528 

529 :param control_scale: bool 

530 if `True`, add scale information in the bottom left corner of the visualization. The default is `True`. 

531 

532 Returns 

533 ------- 

534 `folium.Map` object with the plotted flows. 

535 

536 """ 

537 if map_f is None: 

538 # initialise map 

539 lon, lat = np.mean(np.array(list(fdf.tessellation.geometry.apply(utils.get_geom_centroid).values)), axis=0) 

540 map_f = folium.Map(location=[lat,lon], tiles=tiles, zoom_start=zoom, control_scale=control_scale) 

541 

542 mean_flows = fdf[constants.FLOW].mean() 

543 

544 O_groups = fdf.groupby(by=constants.ORIGIN) 

545 for O, OD in O_groups: 

546 

547 geom = fdf.get_geometry(O) 

548 lonO, latO = utils.get_geom_centroid(geom) 

549

552 continue

554 continue 

555 

556 geom = fdf.get_geometry(D)

560

562 name='geojson', 

563 style_function = style_function(T / mean_flows, flow_color, opacity, 

564 flow_weight, flow_exp) 

565 )

567 popup = folium.Popup('flow from %s to %s: %s'%(O, D, int(T)), max_width=300) 

568 fgeojson = fgeojson.add_child(popup) 

569 

570 fgeojson.add_to(map_f) 

571 

572 if radius_origin_point > 0:

574

577 trips_info = '<br/>'.join(["flow to %s: %s" % 

578 (dd.replace('\'', '_'), int(tt)) \ 

579 for tt, dd in sorted(T_D, reverse=True)[:num_od_popup]]) 

580 

581 geom = fdf.get_geometry(O) 

582 lonO, latO = utils.get_geom_centroid(geom) 

583 fmarker = folium.CircleMarker([latO, lonO], 

584 radius=radius_origin_point, 

585 weight=2, 

586 color=color_origin_point, 

587 fill=True, fill_color=color_origin_point 

588 ) 

589 if tile_popup: 

590 popup = folium.Popup(name+'<br/>'+trips_info, max_width=300) 

591 fmarker = fmarker.add_child(popup) 

592 fmarker.add_to(map_f) 

593 

594 return map_f

617def add_to_map(gway, g, map_f, style_func_args, popup_features=[]): 

618 

619 styles = [] 

620 for k in ['weight', 'color', 'opacity', 'fillColor', 'fillOpacity', 'radius']: 

621 if k in style_func_args: 

622 if callable(style_func_args[k]): 

623 styles += [style_func_args[k](g)] 

624 else: 

625 styles += [style_func_args[k]] 

626 else: 

627 styles += [default_style_func_args[k]] 

628 weight, color, opacity, fillColor, fillOpacity, radius = styles 

629 

630 color, fillColor = manage_colors(color, fillColor) 

631 

632 if type(gway) == shapely.geometry.multipolygon.MultiPolygon: 

633 

634 # Multipolygon 

635 vertices = [list(zip(*p.exterior.xy)) for p in gway] 

636 gj = folium.GeoJson({"type": "MultiPolygon", "coordinates": [vertices]}, 

637 style_function=geojson_style_function(weight=weight, color=color, opacity=opacity, 

638 fillColor=fillColor, fillOpacity=fillOpacity)) 

639 

640 elif type(gway) == shapely.geometry.polygon.Polygon: 

641 

642 # Polygon 

643 vertices = list(zip(*gway.exterior.xy)) 

644 gj = folium.GeoJson({"type": "Polygon", "coordinates": [vertices]}, 

645 style_function=geojson_style_function(weight=weight, color=color, opacity=opacity, 

646 fillColor=fillColor, fillOpacity=fillOpacity)) 

647 

648 elif type(gway) == shapely.geometry.multilinestring.MultiLineString: 

649 

650 # MultiLine 

651 vertices = [list(zip(*l.xy)) for l in gway] 

652 gj = folium.GeoJson({"type": "MultiLineString", "coordinates": vertices}, 

653 style_function=geojson_style_function(weight=weight, color=color, opacity=opacity, 

654 fillColor=fillColor, fillOpacity=fillOpacity)) 

655 

656 elif type(gway) == shapely.geometry.linestring.LineString: 

657 

658 # LineString 

659 vertices = list(zip(*gway.xy)) 

660 gj = folium.GeoJson({"type": "LineString", "coordinates": vertices}, 

661 style_function=geojson_style_function(weight=weight, color=color, opacity=opacity, 

662 fillColor=fillColor, fillOpacity=fillOpacity)) 

663 

664 else: 

665 

666 # Point 

667 point = list(zip(*gway.xy))[0] 

668 # gj = folium.CircleMarker( 

669 gj = folium.Circle( 

670 location=point[::-1], 

671 radius=radius, 

672 color=color, # '#3186cc', 

673 fill=True, 

674 fill_color=fillColor 

675 ) 

676 

677 popup = [] 

678 for pf in popup_features: 

679 try: 

680 popup += ['%s: %s' % (pf, g[pf])] 

681 except KeyError: 

682 pass 

683 

684 try: 

685 popup = '<br>'.join(popup) 

686 popup += json.dumps(g.tags) 

687 popup = popup.replace("""'""", """_""") 

688 except AttributeError: 

689 pass 

690 if len(popup) > 0: 

691 gj.add_child(folium.Popup(popup, max_width=300)) 

692 

693 gj.add_to(map_f) 

694 

695 return map_f

698def plot_gdf(gdf, map_f=None, maxitems=-1, style_func_args={}, popup_features=[], 

699 tiles='cartodbpositron', zoom=6, geom_col='geometry', control_scale=True): 

700 

701 """ 

702 :param gdf: GeoDataFrame 

703 GeoDataFrame to visualize. 

704 

705 :param map_f: folium.Map 

706 `folium.Map` object where the GeoDataFrame `gdf` will be plotted. If `None`, a new map will be created. 

707 

708 :param maxitems: int 

709 maximum number of tiles to plot. If `-1`, all tiles will be plotted. 

710 

711 :param style_func_args: dict 

712 dictionary to pass the following style parameters (keys) to the GeoJson style function of the polygons: 

713 'weight', 'color', 'opacity', 'fillColor', 'fillOpacity', 'radius' 

714 

715 :param popup_features: list 

716 when clicking on a tile polygon, a popup window displaying the information in the 

717 columns of `gdf` listed in `popup_features` will appear. 

718 

719 :param tiles: str 

720 folium's `tiles` parameter. 

721 

722 :param zoom: int 

723 initial zoom. 

724 

725 :param geom_col: str 

726 name of the geometry column of `gdf`. 

727 

728 :param control_scale: bool 

729 if `True`, add scale information in the bottom left corner of the visualization. The default is `True`. 

730 

731 Returns 

732 ------- 

733 `folium.Map` object with the plotted GeoDataFrame. 

734 

735 """ 

736 if map_f is None: 

737 # initialise map 

738 lon, lat = np.mean(np.array(list(gdf[geom_col].apply(utils.get_geom_centroid).values)), axis=0) 

739 map_f = folium.Map(location=[lat, lon], tiles=tiles, zoom_start=zoom, control_scale=control_scale) 

740 

741 

742 

743 count = 0 

744 for k in gdf.index: 

745 g = gdf.loc[k] 

746 

747 if type(g[geom_col]) == gpd.geoseries.GeoSeries: 

748 for i in range(len(g[geom_col])): 

749 map_f = add_to_map(g[geom_col].iloc[i], g.iloc[i], map_f, 

750 popup_features=popup_features, 

751 style_func_args=style_func_args) 

752 else: 

753 map_f = add_to_map(g[geom_col], g, map_f, 

754 popup_features=popup_features, 

755 style_func_args=style_func_args) 

756 

757 count += 1 

758 if count == maxitems: 

759 break 

760 

761 return map_f
